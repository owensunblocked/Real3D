// ---------- NETWORK (PHOTON STYLE) ---------- let socket = new WebSocket("wss://echo.websocket.events"); // demo server let playerId = Math.random().toString(36).substr(2,9); let networkObjects = {}; socket.onmessage = e => { let data = JSON.parse(e.data); if(data.player === playerId) return; let obj = networkObjects[data.id]; if(obj){ obj.position.fromArray(data.pos); obj.rotation.fromArray(data.rot); } }; // ---------- AUTO ADD NETWORK COMPONENT ---------- function addNetworkComponent(obj){ obj.networkId = Math.random().toString(36).substr(2,9); obj.isGrabbed = false; networkObjects[obj.networkId] = obj; } // ---------- OVERRIDE CREATE OBJECT ---------- let oldCreate = createObject; createObject = function(mesh){ oldCreate(mesh); addNetworkComponent(mesh); }; // ---------- SYNC ---------- function syncObject(obj){ if(socket.readyState !== 1) return; socket.send(JSON.stringify({ player: playerId, id: obj.networkId, pos: [obj.position.x,obj.position.y,obj.position.z], rot: [obj.rotation.x,obj.rotation.y,obj.rotation.z] })); } // ---------- VR GRABBING ---------- let grabbed = null; controller1.addEventListener("selectstart", ()=>{ let obj = getClosestObject(controller1.position); if(obj){ grabbed = obj; obj.isGrabbed = true; } }); controller1.addEventListener("selectend", ()=>{ if(grabbed){ grabbed.isGrabbed = false; grabbed = null; } }); // ---------- FIND OBJECT ---------- function getClosestObject(pos){ let closest=null, dist=999; objects.forEach(o=>{ let d = o.position.distanceTo(pos); if(d < dist && d < 1){ dist = d; closest = o; } }); return closest; } // ---------- UPDATE LOOP PATCH ---------- let oldLoop = animate; animate = function(){ requestAnimationFrame(animate); objects.forEach(o=>{ // follow controller if grabbed if(o.isGrabbed){ o.position.copy(controller1.position); } syncObject(o); }); renderer.render(scene,camera); };
